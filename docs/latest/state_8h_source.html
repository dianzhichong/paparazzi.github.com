00501 static inline void stateSetLocalOrigin_i(struct LtpDef_i* ltp_def) {
00502   LTP_DEF_COPY(state.ned_origin_i, *ltp_def);
00503   state.ned_initialized_i = TRUE;
00504   ECEF_FLOAT_OF_BFP(state.ned_origin_f.ecef, state.ned_origin_i.ecef);
00505   LLA_FLOAT_OF_BFP(state.ned_origin_f.lla, state.ned_origin_i.lla);
00506   RMAT_FLOAT_OF_BFP(state.ned_origin_f.ltp_of_ecef, state.ned_origin_i.ltp_of_ecef);
00507   state.ned_origin_f.hmsl = M_OF_MM(state.ned_origin_f.hmsl);
00508   state.ned_initialized_f = TRUE;
00509 
00510   /* clear bits for all local frame representations */
00511   ClearBit(state.pos_status, POS_NED_I);
00512   ClearBit(state.pos_status, POS_ENU_I);
00513   ClearBit(state.pos_status, POS_NED_F);
00514   ClearBit(state.pos_status, POS_ENU_F);
00515   ClearBit(state.speed_status, SPEED_NED_I);
00516   ClearBit(state.speed_status, SPEED_ENU_I);
00517   ClearBit(state.speed_status, SPEED_NED_F);
00518   ClearBit(state.speed_status, SPEED_ENU_F);
00519   ClearBit(state.accel_status, ACCEL_NED_I);
00520   ClearBit(state.accel_status, ACCEL_NED_F);
00521 }
