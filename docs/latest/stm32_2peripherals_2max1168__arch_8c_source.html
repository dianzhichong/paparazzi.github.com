00069 
00070   /* select max1168 */
00071   Max1168Select();
00072 
00073   /* write control byte - wait EOC on extint */
00074   /* use internal reference and clock, sequentially scan channels 0-7 */

00076   spi_write(SPI2, ctl_byte);
00077   max1168_status = STA_MAX1168_SENDING_REQ;
