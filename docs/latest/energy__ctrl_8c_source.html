00192 void v_ctl_altitude_loop( void )
00193 {
00194   // Imput Checks
00195   if (v_ctl_auto_airspeed_setpoint <= 0.0f) v_ctl_auto_airspeed_setpoint = NOMINAL_AIRSPEED;
00196 
00197   // Altitude Controller
00198   v_ctl_altitude_error = v_ctl_altitude_setpoint - stateGetPositionUtm_f()->alt;
00199   float sp = v_ctl_altitude_pgain * v_ctl_altitude_error + v_ctl_altitude_pre_climb ;
00200   BoundAbs(sp, v_ctl_max_climb);

00202   float incr = sp - v_ctl_climb_setpoint;
00203   BoundAbs(incr, 2.0 / 4.0);

00229 void v_ctl_climb_loop( void )
00230 {
00231   // Airspeed outerloop: positive means we need to accelerate
00232   float speed_error = v_ctl_auto_airspeed_setpoint - (*stateGetAirspeed_f());
00233 
00234   // Speed Controller to PseudoControl: gain 1 -> 5m/s error = 0.5g acceleration
00235   v_ctl_desired_acceleration = speed_error * v_ctl_airspeed_pgain / 9.81f;
00236   BoundAbs(v_ctl_desired_acceleration, v_ctl_max_acceleration);
00237 
00238   // Actual Acceleration from IMU: attempt to reconstruct the actual kinematic acceleration
00239 #ifndef SITL
00240   struct Int32Vect3 accel_meas_body;
00241   INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
00242   float vdot = ACCEL_FLOAT_OF_BFP(accel_meas_body.x) / 9.81f - sinf(stateGetNedToBodyEulers_f()->theta);
00243 #else
00244   float vdot = 0;
00245 #endif
00246 
00247   // Acceleration Error: positive means UAV needs to accelerate: needs extra energy
00248   float vdot_err = low_pass_vdot( v_ctl_desired_acceleration - vdot );
00249 
00250   // Flight Path Outerloop: positive means needs to climb more: needs extra energy
00251   float gamma_err  = (v_ctl_climb_setpoint - stateGetSpeedEnu_f()->z) / v_ctl_auto_airspeed_setpoint;
00252 
00253   // Total Energy Error: positive means energy should be added
00254   float en_tot_err = gamma_err + vdot_err;
00255 
00256   // Energy Distribution Error: positive means energy should go from overspeed to altitude = pitch up
00257   float en_dis_err = gamma_err - vdot_err;
00258 
00259   // Auto Cruise Throttle
00260   if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
00261   {
00262     v_ctl_auto_throttle_nominal_cruise_throttle +=
00263             v_ctl_auto_throttle_of_airspeed_igain * speed_error * dt
00264     + en_tot_err * v_ctl_energy_total_igain * dt;
00265     if (v_ctl_auto_throttle_nominal_cruise_throttle < 0.0f) v_ctl_auto_throttle_nominal_cruise_throttle = 0.0f;
00266     else if (v_ctl_auto_throttle_nominal_cruise_throttle > 1.0f) v_ctl_auto_throttle_nominal_cruise_throttle = 1.0f;
00267   }
00268 
00269   // Total Controller
00270   float controlled_throttle = v_ctl_auto_throttle_nominal_cruise_throttle
00271     + v_ctl_auto_throttle_climb_throttle_increment * v_ctl_climb_setpoint
00272     + v_ctl_auto_throttle_of_airspeed_pgain * speed_error
00273     + v_ctl_energy_total_pgain * en_tot_err;
00274 
00275 
00276   if ((controlled_throttle >= 1.0f) || (controlled_throttle <= 0.0f))
00277   {
00278     // If your energy supply is not sufficient, then neglect the climb requirement
00279     en_dis_err = -vdot_err;
00280   }
00281 
00282   /* pitch pre-command */
00283   if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
00284   {
00285     v_ctl_auto_throttle_nominal_cruise_pitch +=  v_ctl_auto_pitch_of_airspeed_igain * (-speed_error) * dt
00286                           + v_ctl_energy_diff_igain * en_dis_err * dt;
00287     Bound(v_ctl_auto_throttle_nominal_cruise_pitch,H_CTL_PITCH_MIN_SETPOINT, H_CTL_PITCH_MAX_SETPOINT);
00288   }
00289   float v_ctl_pitch_of_vz =
00290     + (v_ctl_climb_setpoint /*+ d_err * v_ctl_auto_throttle_pitch_of_vz_dgain*/) * v_ctl_auto_throttle_pitch_of_vz_pgain
00291     - v_ctl_auto_pitch_of_airspeed_pgain * speed_error
00292                 + v_ctl_auto_pitch_of_airspeed_dgain * vdot
00293                 + v_ctl_energy_diff_pgain * en_dis_err
00294                 + v_ctl_auto_throttle_nominal_cruise_pitch;
00295 
00296   v_ctl_pitch_setpoint = v_ctl_pitch_of_vz;
00297 
00298   ac_char_update(controlled_throttle, v_ctl_pitch_of_vz, v_ctl_climb_setpoint, v_ctl_desired_acceleration);
00299 
00300   v_ctl_throttle_setpoint = TRIM_UPPRZ(controlled_throttle * MAX_PPRZ);
