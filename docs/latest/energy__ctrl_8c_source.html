00226 {

00262 void v_ctl_climb_loop( void )
00263 {
00264   // Airspeed outerloop: positive means we need to accelerate
00265   float speed_error = v_ctl_auto_airspeed_setpoint - (*stateGetAirspeed_f());
00266 
00267   // Speed Controller to PseudoControl: gain 1 -> 5m/s error = 0.5g acceleration
00268   v_ctl_desired_acceleration = speed_error * v_ctl_airspeed_pgain / 9.81f;
00269   BoundAbs(v_ctl_desired_acceleration, v_ctl_max_acceleration);
00270 
00271   // Actual Acceleration from IMU: attempt to reconstruct the actual kinematic acceleration
00272 #ifndef SITL
00273   struct Int32Vect3 accel_meas_body;
00274   INT32_RMAT_TRANSP_VMULT(accel_meas_body, imu.body_to_imu_rmat, imu.accel);
00275   float vdot = ACCEL_FLOAT_OF_BFP(accel_meas_body.x) / 9.81f - sinf(stateGetNedToBodyEulers_f()->theta);
00276 #else
00277   float vdot = 0;
00278 #endif
00279 
00280   // Acceleration Error: positive means UAV needs to accelerate: needs extra energy
00281   float vdot_err = low_pass_vdot( v_ctl_desired_acceleration - vdot );
00282 
00283   // Flight Path Outerloop: positive means needs to climb more: needs extra energy
00284   float gamma_err  = (v_ctl_climb_setpoint - stateGetSpeedEnu_f()->z) / v_ctl_auto_airspeed_setpoint;
00285 
00286   // Total Energy Error: positive means energy should be added
00287   float en_tot_err = gamma_err + vdot_err;
00288 
00289   // Energy Distribution Error: positive means energy should go from overspeed to altitude = pitch up
00290   float en_dis_err = gamma_err - vdot_err;
00291 
00292   // Auto Cruise Throttle
00293   if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
00294   {
00295     v_ctl_auto_throttle_nominal_cruise_throttle +=
00296             v_ctl_auto_throttle_of_airspeed_igain * speed_error * dt
00297     + en_tot_err * v_ctl_energy_total_igain * dt;
00298     if (v_ctl_auto_throttle_nominal_cruise_throttle < 0.0f) v_ctl_auto_throttle_nominal_cruise_throttle = 0.0f;
00299     else if (v_ctl_auto_throttle_nominal_cruise_throttle > 1.0f) v_ctl_auto_throttle_nominal_cruise_throttle = 1.0f;
00300   }
00301 
00302   // Total Controller
00303   float controlled_throttle = v_ctl_auto_throttle_nominal_cruise_throttle
00304     + v_ctl_auto_throttle_climb_throttle_increment * v_ctl_climb_setpoint
00305     + v_ctl_auto_throttle_of_airspeed_pgain * speed_error
00306     + v_ctl_energy_total_pgain * en_tot_err;
00307 
00308 
00309   if ((controlled_throttle >= 1.0f) || (controlled_throttle <= 0.0f))
00310   {
00311     // If your energy supply is not sufficient, then neglect the climb requirement
00312     en_dis_err = -vdot_err;
00313   }
00314 
00315   /* pitch pre-command */
00316   if (launch && (v_ctl_mode >= V_CTL_MODE_AUTO_CLIMB))
00317   {
00318     v_ctl_auto_throttle_nominal_cruise_pitch +=  v_ctl_auto_pitch_of_airspeed_igain * (-speed_error) * dt
00319                           + v_ctl_energy_diff_igain * en_dis_err * dt;
00320     Bound(v_ctl_auto_throttle_nominal_cruise_pitch,H_CTL_PITCH_MIN_SETPOINT, H_CTL_PITCH_MAX_SETPOINT);
00321   }
00322   float v_ctl_pitch_of_vz =
00323     + (v_ctl_climb_setpoint /*+ d_err * v_ctl_auto_throttle_pitch_of_vz_dgain*/) * v_ctl_auto_throttle_pitch_of_vz_pgain
00324     - v_ctl_auto_pitch_of_airspeed_pgain * speed_error
00325                 + v_ctl_auto_pitch_of_airspeed_dgain * vdot
00326                 + v_ctl_energy_diff_pgain * en_dis_err
00327                 + v_ctl_auto_throttle_nominal_cruise_pitch;
00328 
00329   v_ctl_pitch_setpoint = v_ctl_pitch_of_vz;
00330 
00331   ac_char_update(controlled_throttle, v_ctl_pitch_of_vz, v_ctl_climb_setpoint, v_ctl_desired_acceleration);
00332 
00333   v_ctl_throttle_setpoint = TRIM_UPPRZ(controlled_throttle * MAX_PPRZ);
00334 }
