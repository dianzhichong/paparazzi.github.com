00193 static inline void autopilot_check_in_flight( void) {
00194   if (autopilot_in_flight) {
00195     if (autopilot_in_flight_counter > 0) {
00196       if (THROTTLE_STICK_DOWN()) {
00197         autopilot_in_flight_counter--;
00198         if (autopilot_in_flight_counter == 0) {
00199           autopilot_in_flight = FALSE;
00200         }
00201       }
00202       else {  /* !THROTTLE_STICK_DOWN */
00203         autopilot_in_flight_counter = AUTOPILOT_IN_FLIGHT_TIME;
00204       }
00205     }
00206   }
00207   else { /* not in flight */
00208     if (autopilot_in_flight_counter < AUTOPILOT_IN_FLIGHT_TIME &&
00209         autopilot_motors_on) {
00210       if (!THROTTLE_STICK_DOWN()) {
00211         autopilot_in_flight_counter++;
00212         if (autopilot_in_flight_counter == AUTOPILOT_IN_FLIGHT_TIME)
00213           autopilot_in_flight = TRUE;
00214       }
00215       else { /*  THROTTLE_STICK_DOWN */
00216         autopilot_in_flight_counter = 0;
00217       }
00218     }
00219   }
00220 }

00248 static inline void autopilot_check_motors_on( void ) {
00249   switch(autopilot_check_motor_status) {
00250     case STATUS_MOTORS_OFF:
00251       autopilot_motors_on = FALSE;
00252       autopilot_motors_on_counter = 0;
00253       if (THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED()) // stick pushed
00254         autopilot_check_motor_status = STATUS_M_OFF_STICK_PUSHED;
00255       break;
00256     case STATUS_M_OFF_STICK_PUSHED:
00257       autopilot_motors_on = FALSE;
00258       autopilot_motors_on_counter++;
00259       if (autopilot_motors_on_counter >= AUTOPILOT_MOTOR_ON_TIME)
00260         autopilot_check_motor_status = STATUS_START_MOTORS;
00261       else if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) // stick released too soon
00262         autopilot_check_motor_status = STATUS_MOTORS_OFF;
00263       break;
00264     case STATUS_START_MOTORS:
00265       autopilot_motors_on = TRUE;
00266       autopilot_motors_on_counter = AUTOPILOT_MOTOR_ON_TIME;
00267       if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) // wait until stick released
00268         autopilot_check_motor_status = STATUS_MOTORS_ON;
00269       break;
00270     case STATUS_MOTORS_ON:
00271       autopilot_motors_on = TRUE;
00272       autopilot_motors_on_counter = AUTOPILOT_MOTOR_ON_TIME;
00273       if (THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED()) // stick pushed
00274         autopilot_check_motor_status = STATUS_M_ON_STICK_PUSHED;
00275       break;
00276     case STATUS_M_ON_STICK_PUSHED:
00277       autopilot_motors_on = TRUE;
00278       autopilot_motors_on_counter--;
00279       if (autopilot_motors_on_counter == 0)
00280         autopilot_check_motor_status = STATUS_STOP_MOTORS;
00281       else if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) // stick released too soon
00282         autopilot_check_motor_status = STATUS_MOTORS_ON;
00283       break;
00284     case STATUS_STOP_MOTORS:
00285       autopilot_motors_on = FALSE;
00286       autopilot_motors_on_counter = 0;
00287       if (!(THROTTLE_STICK_DOWN() && YAW_STICK_PUSHED())) // wait until stick released
00288         autopilot_check_motor_status = STATUS_MOTORS_OFF;
00289       break;
00290     default:
00291       break;
00292   };
00293 }
