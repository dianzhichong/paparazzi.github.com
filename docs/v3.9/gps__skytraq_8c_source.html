00125 void gps_skytraq_parse(uint8_t c) {
00126   if (gps_skytraq.status < GOT_PAYLOAD)
00127     gps_skytraq.checksum ^= c;
00128   switch (gps_skytraq.status) {
00129   case UNINIT:
00130     if (c == SKYTRAQ_SYNC1)
00131       gps_skytraq.status = GOT_SYNC1;
00132     break;
00133   case GOT_SYNC1:
00134     if (c != SKYTRAQ_SYNC2) {
00135       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
00136       goto error;
00137     }
00138     gps_skytraq.status = GOT_SYNC2;
00139     break;
00140   case GOT_SYNC2:
00141     gps_skytraq.len = c<<8;
00142     gps_skytraq.status = GOT_LEN1;
00143     break;
00144   case GOT_LEN1:
00145     gps_skytraq.len += c;
00146     gps_skytraq.status = GOT_LEN2;
00147     if (gps_skytraq.len > GPS_SKYTRAQ_MAX_PAYLOAD) {
00148       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_MSG_TOO_LONG;
00149       goto error;
00150     }
00151     break;
00152   case GOT_LEN2:
00153     gps_skytraq.msg_id = c;
00154     gps_skytraq.msg_idx = 0;
00155     gps_skytraq.checksum = c;
00156     gps_skytraq.status = GOT_ID;
00157     break;
00158   case GOT_ID:
00159     gps_skytraq.msg_buf[gps_skytraq.msg_idx] = c;
00160     gps_skytraq.msg_idx++;
00161     if (gps_skytraq.msg_idx >= gps_skytraq.len-1) {
00162       gps_skytraq.status = GOT_PAYLOAD;
00163     }
00164     break;
00165   case GOT_PAYLOAD:
00166     if (c != gps_skytraq.checksum) {
00167       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_CHECKSUM;
00168       goto error;
00169     }
00170     gps_skytraq.status = GOT_CHECKSUM;
00171     break;
00172   case GOT_CHECKSUM:
00173     if (c != SKYTRAQ_SYNC3) {
00174       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
00175       goto error;
00176     }
00177     gps_skytraq.status = GOT_SYNC3;
00178     break;
00179   case GOT_SYNC3:
00180     gps_skytraq.msg_available = TRUE;
00181     goto restart;
00182   default:
00183     gps_skytraq.error_last = GPS_SKYTRAQ_ERR_UNEXPECTED;
00184     goto error;
00185   }
00186   return;
00187  error:
00188   gps_skytraq.error_cnt++;
00189  restart:
00190   gps_skytraq.status = UNINIT;
00191   return;
00192 }
