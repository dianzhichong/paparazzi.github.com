00144 static inline void parse_extra_pprz( uint8_t c ) {
00145   static uint8_t pprz_status = UNINIT;
00146   static uint8_t _ck_a, _ck_b, payload_idx;
00147 
00148   //uint8_t tab[] = { c };
00149   //DOWNLINK_SEND_DEBUG(DefaultChannel,1,tab);
00150   switch (pprz_status) {
00151   case UNINIT:
00152     if (c == STX)
00153       pprz_status++;
00154     break;
00155   case GOT_STX:
00156     if (extra_pprz_msg_received) {
00157       extra_pprz_ovrn++;
00158       goto error;
00159     }
00160     extra_pprz_payload_len = c-4; /* Counting STX, LENGTH and CRC1 and CRC2 */
00161     _ck_a = _ck_b = c;
00162     pprz_status++;
00163     payload_idx = 0;
00164     break;
00165   case GOT_LENGTH:
00166     extra_pprz_payload[payload_idx] = c;
00167     _ck_a += c; _ck_b += _ck_a;
00168     payload_idx++;
00169     if (payload_idx == extra_pprz_payload_len)
00170       pprz_status++;
00171     break;
00172   case GOT_PAYLOAD:
00173     if (c != _ck_a)
00174       goto error;
00175     pprz_status++;
00176     break;
00177   case GOT_CRC1:
00178     if (c != _ck_b)
00179       goto error;
00180     extra_pprz_msg_received = TRUE;
00181     goto restart;
00182   default:
00183     goto error;
00184   }
00185   return;
00186  error:
00187   extra_pprz_error++;
00188  restart:
00189   pprz_status = UNINIT;
00190   return;
00191 }
