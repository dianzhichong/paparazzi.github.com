00224 __attribute__ ((always_inline)) static inline void  guidance_h_hover_run(void) {
00225 
00226   /* compute position error    */
00227   VECT2_DIFF(guidance_h_pos_err, ins_ltp_pos, guidance_h_pos_sp);
00228   /* saturate it               */
00229   VECT2_STRIM(guidance_h_pos_err, -MAX_POS_ERR, MAX_POS_ERR);
00230 
00231   /* compute speed error    */
00232   VECT2_COPY(guidance_h_speed_err, ins_ltp_speed);
00233   /* saturate it               */
00234   VECT2_STRIM(guidance_h_speed_err, -MAX_SPEED_ERR, MAX_SPEED_ERR);
00235 
00236   /* update pos error integral */
00237   VECT2_ADD(guidance_h_pos_err_sum, guidance_h_pos_err);
00238   /* saturate it               */
00239   VECT2_STRIM(guidance_h_pos_err_sum, -MAX_POS_ERR_SUM, MAX_POS_ERR_SUM);
00240 
00241   /* run PID */
00242   // cmd_earth < 15.17
00243   guidance_h_command_earth.x = (guidance_h_pgain<<1)  * guidance_h_pos_err.x +
00244                                      guidance_h_dgain * (guidance_h_speed_err.x>>9) +
00245                                       guidance_h_igain * (guidance_h_pos_err_sum.x >> 12);
00246   guidance_h_command_earth.y = (guidance_h_pgain<<1)  * guidance_h_pos_err.y +
00247                                      guidance_h_dgain *( guidance_h_speed_err.y>>9) +
00248                           guidance_h_igain * (guidance_h_pos_err_sum.y >> 12);
00249 
00250   VECT2_STRIM(guidance_h_command_earth, -MAX_BANK, MAX_BANK);
00251 
00252   /* Rotate to body frame */
00253   int32_t s_psi, c_psi;
00254   PPRZ_ITRIG_SIN(s_psi, ahrs.ltp_to_body_euler.psi);
00255   PPRZ_ITRIG_COS(c_psi, ahrs.ltp_to_body_euler.psi);
00256 
00257 
00258   // INT32_TRIG_FRAC - 2: 100mm erreur, gain 100 -> 10000 command | 2 degres = 36000, so multiply by 4
00259   guidance_h_command_body.phi =
00260       ( - s_psi * guidance_h_command_earth.x + c_psi * guidance_h_command_earth.y)
00261     >> (INT32_TRIG_FRAC - 2);
00262   guidance_h_command_body.theta =
00263     - ( c_psi * guidance_h_command_earth.x + s_psi * guidance_h_command_earth.y)
00264     >> (INT32_TRIG_FRAC - 2);
00265 
00266 
00267   guidance_h_command_body.phi   += guidance_h_rc_sp.phi;
00268   guidance_h_command_body.theta += guidance_h_rc_sp.theta;
00269   guidance_h_command_body.psi    = guidance_h_psi_sp + guidance_h_rc_sp.psi;
00270 #ifndef STABILISATION_ATTITUDE_TYPE_FLOAT
00271   ANGLE_REF_NORMALIZE(guidance_h_command_body.psi);
00272 #endif /* STABILISATION_ATTITUDE_TYPE_FLOAT */
00273 
00274   EULERS_COPY(stab_att_sp_euler, guidance_h_command_body);
00275 
00276 }

00282 __attribute__ ((always_inline)) static inline void  guidance_h_nav_run(bool_t in_flight) {
00283 
00284   /* convert our reference to generic representation */
00285 #ifdef GUIDANCE_H_USE_REF
00286   INT32_VECT2_RSHIFT(guidance_h_pos_ref,   b2_gh_pos_ref,   (B2_GH_POS_REF_FRAC - INT32_POS_FRAC));
00287   INT32_VECT2_LSHIFT(guidance_h_speed_ref, b2_gh_speed_ref, (INT32_SPEED_FRAC - B2_GH_SPEED_REF_FRAC));
00288   INT32_VECT2_LSHIFT(guidance_h_accel_ref, b2_gh_accel_ref, (INT32_ACCEL_FRAC - B2_GH_ACCEL_REF_FRAC));
00289 #else
00290   VECT2_COPY(guidance_h_pos_ref, guidance_h_pos_sp);
00291   INT_VECT2_ZERO(guidance_h_speed_ref);
00292   INT_VECT2_ZERO(guidance_h_accel_ref);
00293 #endif
00294 
00295   /* compute position error    */
00296   VECT2_DIFF(guidance_h_pos_err, ins_ltp_pos, guidance_h_pos_ref);
00297   /* saturate it               */
00298   VECT2_STRIM(guidance_h_pos_err, -MAX_POS_ERR, MAX_POS_ERR);
00299 
00300   /* compute speed error    */
00301   //VECT2_COPY(guidance_h_speed_err, ins_ltp_speed);
00302   VECT2_DIFF(guidance_h_speed_err, ins_ltp_speed, guidance_h_speed_ref);
00303   /* saturate it               */
00304   VECT2_STRIM(guidance_h_speed_err, -MAX_SPEED_ERR, MAX_SPEED_ERR);
00305 
00306   int32_t dist;
00307   INT32_VECT2_NORM(dist, guidance_h_pos_err);
00308   if ( dist < HOLD_DISTANCE ) { // Hold position
00309     /* update pos error integral */
00310     VECT2_ADD(guidance_h_pos_err_sum, guidance_h_pos_err);
00311     /* saturate it               */
00312     VECT2_STRIM(guidance_h_pos_err_sum, -MAX_POS_ERR_SUM, MAX_POS_ERR_SUM);
00313   }
00314   else { // Tracking algorithm, no integral
00315     int32_t vect_prod = 0;
00316     int32_t scal_prod = ins_ltp_speed.x * guidance_h_pos_err.x + ins_ltp_speed.y * guidance_h_pos_err.y;
00317     // compute vectorial product only if angle < pi/2 (scalar product > 0)
00318     if (scal_prod >= 0) {
00319       vect_prod = ((ins_ltp_speed.x * guidance_h_pos_err.y) >> (INT32_POS_FRAC + INT32_SPEED_FRAC - 10))
00320                 - ((ins_ltp_speed.y * guidance_h_pos_err.x) >> (INT32_POS_FRAC + INT32_SPEED_FRAC - 10));
00321     }
00322     // multiply by vector orthogonal to speed
00323     VECT2_ASSIGN(guidance_h_nav_err,
00324         vect_prod * (-ins_ltp_speed.y),
00325         vect_prod * ins_ltp_speed.x);
00326     // divide by 2 times dist ( >> 16 )
00327     VECT2_SDIV(guidance_h_nav_err, guidance_h_nav_err, dist*dist);
00328     // *2 ??
00329   }
00330   if (!in_flight) { INT_VECT2_ZERO(guidance_h_pos_err_sum); }
00331 
00332   /* run PID */
00333   guidance_h_command_earth.x =
00334     guidance_h_pgain * (guidance_h_pos_err.x << (10 - INT32_POS_FRAC)) +
00335     guidance_h_dgain * (guidance_h_speed_err.x >> (INT32_SPEED_FRAC - 10)) +
00336     guidance_h_igain * (guidance_h_pos_err_sum.x >> (12 + INT32_POS_FRAC - 10)) +
00337     guidance_h_ngain * guidance_h_nav_err.x +
00338     guidance_h_again * guidance_h_accel_ref.x; /* feedforward gain */
00339   guidance_h_command_earth.y =
00340     guidance_h_pgain * (guidance_h_pos_err.y << (10 - INT32_POS_FRAC)) +
00341     guidance_h_dgain * (guidance_h_speed_err.y >> (INT32_SPEED_FRAC - 10)) +
00342     guidance_h_igain * (guidance_h_pos_err_sum.y >> (12 + INT32_POS_FRAC - 10)) +
00343     guidance_h_ngain * guidance_h_nav_err.y +
00344     guidance_h_again * guidance_h_accel_ref.y; /* feedforward gain */
00345 
00346   VECT2_STRIM(guidance_h_command_earth, -NAV_MAX_BANK, NAV_MAX_BANK);
00347   INT32_VECT2_RSHIFT(guidance_h_command_earth, guidance_h_command_earth, REF_ANGLE_FRAC - 16); // Reduice to 16 for trigo operation
00348 
00349   /* Rotate to body frame */
00350   int32_t s_psi, c_psi;
00351   PPRZ_ITRIG_SIN(s_psi, ahrs.ltp_to_body_euler.psi);
00352   PPRZ_ITRIG_COS(c_psi, ahrs.ltp_to_body_euler.psi);
00353 
00354   // Restore angle ref resolution after rotation
00355   guidance_h_command_body.phi =
00356       ( - s_psi * guidance_h_command_earth.x + c_psi * guidance_h_command_earth.y) >> (INT32_TRIG_FRAC - (REF_ANGLE_FRAC - 16));
00357   guidance_h_command_body.theta =
00358     - ( c_psi * guidance_h_command_earth.x + s_psi * guidance_h_command_earth.y) >> (INT32_TRIG_FRAC - (REF_ANGLE_FRAC - 16));
00359 
00360   // Add RC setpoint
00361   guidance_h_command_body.phi   += guidance_h_rc_sp.phi;
00362   guidance_h_command_body.theta += guidance_h_rc_sp.theta;
00363   guidance_h_command_body.psi    = guidance_h_psi_sp + guidance_h_rc_sp.psi;
00364   ANGLE_REF_NORMALIZE(guidance_h_command_body.psi);
00365 
00366   // Set attitude setpoint
00367   EULERS_COPY(stab_att_sp_euler, guidance_h_command_body);
00368 
00369 }
