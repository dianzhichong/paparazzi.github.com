00106 __attribute__ ((always_inline)) static inline void gv_update_ref_from_z_sp(int32_t z_sp) {
00107 
00108   gv_z_ref  += gv_zd_ref;
00109   gv_zd_ref += gv_zdd_ref;
00110 
00111   // compute the "speed part" of zdd = -2*zeta*omega*zd -omega^2(z_sp - z)
00112   int32_t zd_zdd_res = gv_zd_ref>>(GV_ZD_REF_FRAC - GV_ZDD_REF_FRAC);
00113   int32_t zdd_speed = ((int32_t)(-2*GV_ZETA_OMEGA)*zd_zdd_res)>>(GV_ZETA_OMEGA_FRAC);
00114   // compute z error in z_sp resolution
00115   int32_t z_err_sp = z_sp - (int32_t)(gv_z_ref>>(GV_Z_REF_FRAC-INT32_POS_FRAC));
00116   // convert to accel resolution
00117   int32_t z_err_accel = z_err_sp>>(INT32_POS_FRAC-GV_ZDD_REF_FRAC);
00118   int32_t zdd_pos = ((int32_t)(GV_OMEGA_2)*z_err_accel)>>GV_OMEGA_2_FRAC;
00119   gv_zdd_ref = zdd_speed + zdd_pos;
00120 
00121   /* Saturate accelerations */
00122   Bound(gv_zdd_ref, GV_MIN_ZDD, GV_MAX_ZDD);
00123 
00124   /* Saturate speed and adjust acceleration accordingly */
00125   if (gv_zd_ref <= GV_MIN_ZD) {
00126     gv_zd_ref = GV_MIN_ZD;
00127     if (gv_zdd_ref < 0)
00128       gv_zdd_ref = 0;
00129   }
00130   else if (gv_zd_ref >= GV_MAX_ZD) {
00131     gv_zd_ref = GV_MAX_ZD;
