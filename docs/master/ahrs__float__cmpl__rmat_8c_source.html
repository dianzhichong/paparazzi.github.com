00092 
00093   /* converts gyro to floating point */
00094   struct FloatRates gyro_float;
00095   RATES_FLOAT_OF_BFP(gyro_float, imu.gyro_prev);
00096   /* unbias measurement */
00097   RATES_SUB(gyro_float, ahrs_impl.gyro_bias);
00098 
00099 #ifdef AHRS_PROPAGATE_LOW_PASS_RATES
00100   const float alpha = 0.1;
00101   FLOAT_RATES_LIN_CMB(ahrs_float.imu_rate, ahrs_float.imu_rate, (1.-alpha), gyro_float, alpha);
00102 #else
00103   RATES_COPY(ahrs_float.imu_rate,gyro_float);
00104 #endif
00105 
00106   /* add correction     */
00107   struct FloatRates omega;
00108   RATES_SUM(omega, gyro_float, ahrs_impl.rate_correction);
00109   /* and zeros it */
00110   FLOAT_RATES_ZERO(ahrs_impl.rate_correction);
00111 
00112   const float dt = 1./AHRS_PROPAGATE_FREQUENCY;
00113 #ifdef AHRS_PROPAGATE_RMAT
00114   FLOAT_RMAT_INTEGRATE_FI(ahrs_float.ltp_to_imu_rmat, omega, dt );
00115   float_rmat_reorthogonalize(&ahrs_float.ltp_to_imu_rmat);
00116   compute_imu_quat_and_euler_from_rmat();
00117 #endif
00118 #ifdef AHRS_PROPAGATE_QUAT
00119   FLOAT_QUAT_INTEGRATE(ahrs_float.ltp_to_imu_quat, omega, dt);
00120   FLOAT_QUAT_NORMALIZE(ahrs_float.ltp_to_imu_quat);
00121   compute_imu_rmat_and_euler_from_quat();

00140           RMAT_ELMT(ahrs_float.ltp_to_imu_rmat, 2,2)};

00149 #endif
00150   /* compute correction */
00151   const float gravity_rate_update_gain = -5e-2; // -5e-2

00153 #if 1
00154   const float gravity_bias_update_gain = 1e-5; // -5e-6

00226 void ahrs_update_mag_full_2d_dumb(void) {
00227 
00228   /* project mag on local tangeant plane */
00229   struct FloatVect3 magf;
00230   MAGS_FLOAT_OF_BFP(magf, imu.mag);
00231   const float cphi   = cosf(ahrs_float.ltp_to_imu_euler.phi);
00232   const float sphi   = sinf(ahrs_float.ltp_to_imu_euler.phi);
00233   const float ctheta = cosf(ahrs_float.ltp_to_imu_euler.theta);
00234   const float stheta = sinf(ahrs_float.ltp_to_imu_euler.theta);
00235   const float mn = ctheta * magf.x + sphi*stheta*magf.y + cphi*stheta*magf.z;
00236   const float me =     0. * magf.x + cphi       *magf.y - sphi       *magf.z;
00237 
00238   const float res_norm = -RMAT_ELMT(ahrs_float.ltp_to_imu_rmat, 0,0)*me + RMAT_ELMT(ahrs_float.ltp_to_imu_rmat, 1,0)*mn;
00239   struct FloatVect3* r2 = (struct FloatVect3*)(&RMAT_ELMT(ahrs_float.ltp_to_imu_rmat, 2,0));
00240   const float mag_rate_update_gain = 2.5;
00241   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.rate_correction, (*r2), (mag_rate_update_gain*res_norm));
00242   const float mag_bias_update_gain = -2.5e-4;
00243   FLOAT_RATES_ADD_SCALED_VECT(ahrs_impl.gyro_bias, (*r2), (mag_bias_update_gain*res_norm));
