00154 static inline void parse_pprz( uint8_t c ) {
00155   static uint8_t pprz_status = UNINIT;
00156   static uint8_t _ck_a, _ck_b, payload_idx;
00157 
00158   switch (pprz_status) {
00159   case UNINIT:
00160     if (c == STX)
00161       pprz_status++;
00162     break;
00163   case GOT_STX:
00164     if (pprz_msg_received) {
00165       pprz_ovrn++;
00166       goto error;
00167     }
00168     pprz_payload_len = c-4; /* Counting STX, LENGTH and CRC1 and CRC2 */
00169     _ck_a = _ck_b = c;
00170     pprz_status++;
00171     payload_idx = 0;
00172     break;
00173   case GOT_LENGTH:
00174     pprz_payload[payload_idx] = c;
00175     _ck_a += c; _ck_b += _ck_a;
00176     payload_idx++;
00177     if (payload_idx == pprz_payload_len)
00178       pprz_status++;
00179     break;
00180   case GOT_PAYLOAD:
00181     if (c != _ck_a)
00182       goto error;
00183     pprz_status++;
00184     break;
00185   case GOT_CRC1:
00186     if (c != _ck_b)
00187       goto error;
00188     pprz_msg_received = TRUE;
00189     goto restart;
00190   default:
00191     goto error;
00192   }
00193   return;
00194  error:
00195   pprz_error++;
00196  restart:
00197   pprz_status = UNINIT;
00198   return;
00199 }
