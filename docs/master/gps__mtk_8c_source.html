00092 static void gps_mtk_time2itow(uint32_t  gps_date, uint32_t  gps_time,
00093                            int16_t* gps_week, uint32_t* gps_itow) {
00094   /* convert UTC date/time to GPS week/itow, we have no idea about GPS
00095      leap seconds for now */
00096   uint16_t gps_msecond = gps_time % 1000;
00097   uint8_t  gps_second  = (gps_time / 1000) % 100;
00098   uint8_t  gps_minute  = (gps_time / 100000) % 100;
00099   uint8_t  gps_hour    = (gps_time / 10000000) % 100;
00100   uint16_t gps_year    = 2000 + (gps_date % 100);
00101   uint8_t  gps_month   = (gps_date / 100) % 100;
00102   uint8_t  gps_day     = (gps_date / 10000) % 100;
00103   int32_t  i, days;
00104 
00105   *gps_week = 0;
00106   *gps_itow = 0;
00107 
00108   /* sanity checks */
00109   if (gps_month > 12) return;
00110   if (gps_day > (DAYS_MONTH[gps_month] +
00111     ((gps_month == 1) ? isleap(gps_year):0))) return;
00112   if (gps_hour > 23) return;
00113   if (gps_minute > 59) return;
00114   if (gps_second > 59) return;
00115 
00116   /* days since 6-JAN-1980 */
00117   days = -6;
00118   for (i = 1980; i < gps_year; i++) days += (365 + isleap(i));
00119 
00120   /* add days in gps_year */
00121   for (i = 0; i < gps_month-1; i++) {
00122       days += DAYS_MONTH[i] + ((i == 1) ? isleap(gps_year):0);
00123   }
00124   days += gps_day;
00125 
00126   /* convert */
00127   *gps_week = (uint16_t) (days / 7);
00128   *gps_itow = ((days % 7) * SECS_DAY +
00129     gps_hour * SECS_HOUR +
00130     gps_minute * SECS_MINUTE +
00131     gps_second) * 1000 +
00132     gps_msecond;
00133 }

00260 void gps_mtk_parse( uint8_t c ) {
00261   if (gps_mtk.status < GOT_PAYLOAD) {
00262     gps_mtk.ck_a += c;
00263     gps_mtk.ck_b += gps_mtk.ck_a;
00264   }
00265   switch (gps_mtk.status) {
00266   case UNINIT:
00267     if (c == MTK_DIY14_SYNC1)
00268       gps_mtk.status = GOT_SYNC1_14;
00269     if (c == MTK_DIY16_ID)
00270       gps_mtk.msg_class = c;
00271       gps_mtk.status = GOT_SYNC1_16;
00272     break;
00273   /* MTK_DIY_VER_14 */
00274   case GOT_SYNC1_14:
00275     if (c != MTK_DIY14_SYNC2) {
00276       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00277       goto error;
00278     }
00279     if (gps_mtk.msg_available) {
00280       /* Previous message has not yet been parsed: discard this one */
00281       gps_mtk.error_last = GPS_MTK_ERR_OVERRUN;
00282       goto error;
00283     }
00284     gps_mtk.ck_a = 0;
00285     gps_mtk.ck_b = 0;
00286     gps_mtk.status++;
00287     gps_mtk.len = MTK_DIY14_NAV_LENGTH;
00288     break;
00289   case GOT_SYNC2_14:
00290     if (c != MTK_DIY14_ID) {
00291       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00292       goto error;
00293     }
00294     gps_mtk.msg_class = c;
00295     gps_mtk.msg_idx = 0;
00296     gps_mtk.status++;
00297     break;
00298   case GOT_CLASS_14:
00299     if (c != MTK_DIY14_NAV_ID) {
00300       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00301       goto error;
00302     }
00303     gps_mtk.msg_id = c;
00304     gps_mtk.status = GOT_ID;
00305     break;
00306   /* MTK_DIY_VER_16 */
00307   case GOT_SYNC1_16:
00308     if (c != MTK_DIY16_NAV_ID) {
00309       gps_mtk.error_last = GPS_MTK_ERR_OUT_OF_SYNC;
00310       goto error;
00311     }
00312     if (gps_mtk.msg_available) {
00313       /* Previous message has not yet been parsed: discard this one */
00314       gps_mtk.error_last = GPS_MTK_ERR_OVERRUN;
00315       goto error;
00316     }
00317     gps_mtk.msg_id = c;
00318     gps_mtk.ck_a = 0;
00319     gps_mtk.ck_b = 0;
00320     gps_mtk.status++;
00321     break;
00322   case GOT_SYNC2_16:
00323     gps_mtk.len = c;
00324     gps_mtk.msg_idx = 0;
00325     gps_mtk.status = GOT_ID;
00326     break;
00327   case GOT_ID:
00328     gps_mtk.msg_buf[gps_mtk.msg_idx] = c;
00329     gps_mtk.msg_idx++;
00330     if (gps_mtk.msg_idx >= gps_mtk.len) {
00331       gps_mtk.status++;
00332     }
00333     break;
00334   case GOT_PAYLOAD:
00335     if (c != gps_mtk.ck_a) {
00336       gps_mtk.error_last = GPS_MTK_ERR_CHECKSUM;
00337       goto error;
00338     }
00339     gps_mtk.status++;
00340     break;
00341   case GOT_CHECKSUM1:
00342     if (c != gps_mtk.ck_b) {
00343       gps_mtk.error_last = GPS_MTK_ERR_CHECKSUM;
00344       goto error;

00348     break;
00349   default:
00350     gps_mtk.error_last = GPS_MTK_ERR_UNEXPECTED;
00351     goto error;
00352   }
