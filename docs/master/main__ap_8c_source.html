00445 #if _check_periodic_freq_
00446 #error Using HighSpeed Periodic: PERIODIC_FREQUENCY has to be a multiple of 60!
00447 #endif
00448   _60Hz++;
00449   if (_60Hz >= (PERIODIC_FREQUENCY / 60))
00450   {
00451     _60Hz = 0;
00452   }
00453   else
00454   {
00455     return;
00456   }
00457 
00458 
00459   // Rest of the periodic function still runs at 60Hz like always
00460   _20Hz++;
00461   if (_20Hz>=3) _20Hz=0;
00462   _10Hz++;
00463   if (_10Hz>=6) _10Hz=0;
00464   _4Hz++;
00465   if (_4Hz>=15) _4Hz=0;
00466   _1Hz++;
00467   if (_1Hz>=60) _1Hz=0;
00468 
00469   reporting_task();
00470 
00471   if (!_1Hz) {
00472     if (estimator_flight_time) estimator_flight_time++;
00473 #if defined DATALINK || defined SITL
00474     datalink_time++;
00475 #endif
00476 
00477     static uint8_t t = 0;
00478     if (vsupply < LOW_BATTERY_DECIVOLT) t++; else t = 0;
00479     kill_throttle |= (t >= LOW_BATTERY_DELAY);
00480     kill_throttle |= launch && (dist2_to_home > Square(KILL_MODE_DISTANCE));
00481   }
00482 
00483   switch(_4Hz) {
00484   case 0:
00485     estimator_propagate_state();
00486 #ifdef EXTRA_DOWNLINK_DEVICE

00495       launch = TRUE; /* Not set in non auto launch */
00496       DOWNLINK_SEND_TAKEOFF(DefaultChannel, &cpu_time_sec);
00497   default:
00498     break;
00499     }
00500 
00501     break;
00502 
00503 #ifdef USE_GPIO
00504   case 3:
