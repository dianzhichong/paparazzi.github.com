00160 static inline void parse_xbee( uint8_t c ) {
00161   static uint8_t xbee_status = XBEE_UNINIT;
00162   static uint8_t cs, payload_idx;
00163 
00164   switch (xbee_status) {
00165   case XBEE_UNINIT:
00166     if (c == XBEE_START)
00167       xbee_status++;
00168     break;
00169   case XBEE_GOT_START:
00170     if (xbee_msg_received) {
00171       xbee_ovrn++;
00172       goto error;
00173     }
00174     xbee_payload_len = c<<8;
00175     xbee_status++;
00176     break;
00177   case XBEE_GOT_LENGTH_MSB:
00178     xbee_payload_len |= c;
00179     xbee_status++;
00180     payload_idx = 0;
00181     cs=0;
00182     break;
00183   case XBEE_GOT_LENGTH_LSB:
00184     xbee_payload[payload_idx] = c;
00185     cs += c;
00186     payload_idx++;
00187     if (payload_idx == xbee_payload_len)
00188       xbee_status++;
00189     break;
00190   case XBEE_GOT_PAYLOAD:
00191     if (c + cs != 0xff)
00192       goto error;
00193     xbee_msg_received = TRUE;
00194     goto restart;
00195     break;
00196   default:
00197     goto error;
00198   }
00199   return;
00200  error:
00201   xbee_error++;
00202  restart:
00203   xbee_status = XBEE_UNINIT;
00204   return;
00205 }
