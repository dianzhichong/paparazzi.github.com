00138 static inline void imu_aspirin_event(void (* _gyro_handler)(void), void (* _accel_handler)(void), void (* _mag_handler)(void))
00139 {
00140   if (imu_aspirin.status == AspirinStatusUninit) return;
00141 
00142   imu_aspirin_arch_int_disable();
00143   if (imu_aspirin.accel_available) {
00144     imu_aspirin.time_since_last_accel_reading = 0;
00145     imu_aspirin.accel_available = FALSE;
00146     accel_copy_spi();
00147     _accel_handler();
00148   }
00149   imu_aspirin_arch_int_enable();
00150 
00151   // Reset everything if we've been waiting too long
00152   if (imu_aspirin.time_since_last_reading > ASPIRIN_GYRO_TIMEOUT) {
00153     i2c2_er_irq_handler();
00154     gyro_read_i2c();
00155     imu_aspirin.time_since_last_reading = 0;
00156     return;
00157   }
00158 
00159   // Try again later if transaction is in progress
00160   if (imu_aspirin.i2c_trans_gyro.status == I2CTransPending || imu_aspirin.i2c_trans_gyro.status == I2CTransRunning)
00161   {
00162     return;
00163   }
00164 
00165   ImuMagEvent(_mag_handler);
00166 
00167   // Try back later if things are not idle
00168   if ((i2c2.status != I2CIdle) || !i2c_idle(&i2c2)) {
00169     return;
00170   }
00171 
00172   if (imu_aspirin.reading_gyro) {
00173     if (imu_aspirin.i2c_trans_gyro.status == I2CTransSuccess) {
00174       gyro_copy_i2c();
00175       if (imu_aspirin.gyro_available_blaaa) {
00176         imu_aspirin.gyro_available_blaaa = FALSE;
00177         _gyro_handler();
00178       }
00179     }
00180     imu_aspirin.reading_gyro = 0;
00181     return;
00182   }
00183 
00184   // If we're not already waiting for read, and conversion is complete, schedule a read
00185   if (!imu_aspirin.reading_gyro && imu_aspirin_eoc() && i2c2.status == I2CIdle && i2c_idle(&i2c2)) {
00186     if (imu_aspirin.i2c_trans_gyro.status == I2CTransSuccess) {
00187       imu_aspirin.time_since_last_reading = 0;
00188     }
00189     gyro_read_i2c();
00190     return;
00191   }
00192 
00193 }
