00055 void uart_transmit(struct uart_periph* p, uint8_t data ) {
00056 
00057   uint16_t temp = (p->tx_insert_idx + 1) % UART_TX_BUFFER_SIZE;
00058 
00059   if (temp == p->tx_extract_idx)
00060     return;                          // no room
00061 
00062   USART_ITConfig(p->reg_addr, USART_IT_TXE, DISABLE);
00063 
00064   // check if in process of sending data
00065   if (p->tx_running) { // yes, add to queue
00066     p->tx_buf[p->tx_insert_idx] = data;
00067     p->tx_insert_idx = temp;
00068   }
00069   else { // no, set running flag and write to output register
00070     p->tx_running = TRUE;
00071     USART_SendData(p->reg_addr, data);
00072   }
00073 
00074   USART_ITConfig(p->reg_addr, USART_IT_TXE, ENABLE);
00075 
00076 }

00078 static inline void usart_irq_handler(struct uart_periph* p) {
00079 
00080   if(USART_GetITStatus(p->reg_addr, USART_IT_TXE) != RESET){
00081     // check if more data to send
00082     if (p->tx_insert_idx != p->tx_extract_idx) {
00083       USART_SendData(p->reg_addr,p->tx_buf[p->tx_extract_idx]);
00084       p->tx_extract_idx++;
00085       p->tx_extract_idx %= UART_TX_BUFFER_SIZE;
00086     }
00087     else {
00088       p->tx_running = FALSE;   // clear running flag
00089       USART_ITConfig(p->reg_addr, USART_IT_TXE, DISABLE);
00090     }
00091   }
00092 
00093   if(USART_GetITStatus(p->reg_addr, USART_IT_RXNE) != RESET){
00094     uint16_t temp = (p->rx_insert_idx + 1) % UART_RX_BUFFER_SIZE;;
00095     p->rx_buf[p->rx_insert_idx] = USART_ReceiveData(p->reg_addr);
00096     // check for more room in queue
00097     if (temp != p->rx_extract_idx)
00098       p->rx_insert_idx = temp; // update insert index
00099   }
00100 
00101 }
