00062                                         \
00063     _sp.phi =               \
00064       ((int32_t)-radio_control.values[RADIO_ROLL]  * (int32_t)SP_MAX_PHI / MAX_PPRZ) \
00065       << (REF_ANGLE_FRAC - INT32_ANGLE_FRAC);         \
00066     _sp.theta =               \
00067       ((int32_t) radio_control.values[RADIO_PITCH] * (int32_t)SP_MAX_THETA / MAX_PPRZ) \
00068       << (REF_ANGLE_FRAC - INT32_ANGLE_FRAC);         \
00069     if (_inflight) {              \
00070       if (YAW_DEADBAND_EXCEEDED()) {          \
00071     _sp.psi +=              \
00072       ((int32_t)-radio_control.values[RADIO_YAW] * (int32_t)SP_MAX_R / MAX_PPRZ / RC_UPDATE_FREQ) \
00073       << (REF_ANGLE_FRAC - INT32_ANGLE_FRAC);       \
00074     ANGLE_REF_NORMALIZE(_sp.psi);         \
