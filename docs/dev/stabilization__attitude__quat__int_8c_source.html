00141 static void attitude_run_ff(int32_t ff_commands[], struct Int32AttitudeGains *gains, struct Int32Rates *ref_accel)
00142 {
00143   /* Compute feedforward based on reference acceleration */
00144 
00145   ff_commands[COMMAND_ROLL]          = GAIN_PRESCALER_FF * gains->dd.x * RATE_FLOAT_OF_BFP(ref_accel->p) / (1 << 7);
00146   ff_commands[COMMAND_PITCH]         = GAIN_PRESCALER_FF * gains->dd.y * RATE_FLOAT_OF_BFP(ref_accel->q) / (1 << 7);
00147   ff_commands[COMMAND_YAW]           = GAIN_PRESCALER_FF * gains->dd.z * RATE_FLOAT_OF_BFP(ref_accel->r) / (1 << 7);
00148 }

00171 void stabilization_attitude_run(bool_t enable_integrator) {
00172 
00173   /*
00174    * Update reference
00175    */
00176   stabilization_attitude_ref_update();
00177 
00178   /*
00179    * Compute errors for feedback
00180    */
00181 
00182   /* attitude error                          */
00183   struct Int32Quat att_err;
00184   INT32_QUAT_INV_COMP(att_err, ahrs.ltp_to_body_quat, stab_att_ref_quat);
00185   /* wrap it in the shortest direction       */
00186   INT32_QUAT_WRAP_SHORTEST(att_err);
00187   INT32_QUAT_NORMALIZE(att_err);
00188 
00189   /*  rate error                */
00190   struct Int32Rates rate_err;
00191   RATES_DIFF(rate_err, ahrs.body_rate, stab_att_ref_rate);
00192 
00193   /* integrated error */
00194   if (enable_integrator) {
00195     struct Int32Quat new_sum_err, scaled_att_err;
00196     /* update accumulator */
00197     scaled_att_err.qi = att_err.qi;
00198     scaled_att_err.qx = att_err.qx / IERROR_SCALE;
00199     scaled_att_err.qy = att_err.qy / IERROR_SCALE;
00200     scaled_att_err.qz = att_err.qz / IERROR_SCALE;
00201     INT32_QUAT_COMP_INV(new_sum_err, stabilization_att_sum_err_quat, scaled_att_err);
00202     INT32_QUAT_NORMALIZE(new_sum_err);
00203     QUAT_COPY(stabilization_att_sum_err_quat, new_sum_err);
00204     INT32_EULERS_OF_QUAT(stabilization_att_sum_err, stabilization_att_sum_err_quat);
00205   } else {
00206     /* reset accumulator */
00207     INT32_QUAT_ZERO( stabilization_att_sum_err_quat );
00208     INT_EULERS_ZERO( stabilization_att_sum_err );
00209   }
00210 
00211   attitude_run_ff(stabilization_att_ff_cmd, &stabilization_gains, &stab_att_ref_accel);
00212 
00213   attitude_run_fb(stabilization_att_fb_cmd, &stabilization_gains, &att_err, &rate_err, &stabilization_att_sum_err_quat);
00214 
00215   for (int i = COMMAND_ROLL; i <= COMMAND_YAW; i++) {
00216     stabilization_cmd[i] = stabilization_att_fb_cmd[i]+stabilization_att_ff_cmd[i];
00217      Bound(stabilization_cmd[i], -200, 200);
00218   }
00219 }
