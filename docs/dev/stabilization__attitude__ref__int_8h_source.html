00054 #define REF_ANGLE_PI      BFP_OF_REAL(3.1415926535897932384626433832795029, REF_ANGLE_FRAC)

00056 #define ANGLE_REF_NORMALIZE(_a) {                       \
00057     while (_a >  REF_ANGLE_PI)  _a -= REF_ANGLE_TWO_PI; \

00059   }

00072 static inline void stabilization_attitude_read_rc_ref(struct Int32Eulers *sp, bool_t in_flight) {
00073 
00074   sp->phi = ((int32_t)-radio_control.values[RADIO_ROLL]  * SP_MAX_PHI / MAX_PPRZ)
00075     << (REF_ANGLE_FRAC - INT32_ANGLE_FRAC);
00076 
00077   sp->theta = ((int32_t) radio_control.values[RADIO_PITCH] * SP_MAX_THETA / MAX_PPRZ)
00078     << (REF_ANGLE_FRAC - INT32_ANGLE_FRAC);
00079 
00080   if (in_flight) {
00081     if (YAW_DEADBAND_EXCEEDED()) {
00082       sp->psi += ((int32_t)-radio_control.values[RADIO_YAW] * SP_MAX_R / MAX_PPRZ / RC_UPDATE_FREQ)
00083         << (REF_ANGLE_FRAC - INT32_ANGLE_FRAC);
00084       ANGLE_REF_NORMALIZE(sp->psi);
00085     }
00086   }
00087   else { /* if not flying, use current yaw as setpoint */
00088     sp->psi = (ahrs.ltp_to_body_euler.psi << (REF_ANGLE_FRAC - INT32_ANGLE_FRAC));
00089   }
00090 
00091 }
