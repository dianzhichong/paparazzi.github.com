00086 void gps_skytraq_parse(uint8_t c) {
00087   if (gps_skytraq.status < GOT_PAYLOAD)
00088     gps_skytraq.checksum ^= c;
00089   switch (gps_skytraq.status) {
00090   case UNINIT:
00091     if (c == SKYTRAQ_SYNC1)
00092       gps_skytraq.status = GOT_SYNC1;
00093     break;
00094   case GOT_SYNC1:
00095     if (c != SKYTRAQ_SYNC2) {
00096       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
00097       goto error;
00098     }
00099     gps_skytraq.status = GOT_SYNC2;
00100     break;
00101   case GOT_SYNC2:
00102     gps_skytraq.len = c<<8;
00103     gps_skytraq.status = GOT_LEN1;
00104     break;
00105   case GOT_LEN1:
00106     gps_skytraq.len += c;
00107     gps_skytraq.status = GOT_LEN2;
00108     if (gps_skytraq.len > GPS_SKYTRAQ_MAX_PAYLOAD) {
00109       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_MSG_TOO_LONG;
00110       goto error;
00111     }
00112     break;
00113   case GOT_LEN2:
00114     gps_skytraq.msg_id = c;
00115     gps_skytraq.msg_idx = 0;
00116     gps_skytraq.checksum = c;
00117     gps_skytraq.status = GOT_ID;
00118     break;
00119   case GOT_ID:
00120     gps_skytraq.msg_buf[gps_skytraq.msg_idx] = c;
00121     gps_skytraq.msg_idx++;
00122     if (gps_skytraq.msg_idx >= gps_skytraq.len-1) {
00123       gps_skytraq.status = GOT_PAYLOAD;
00124     }
00125     break;
00126   case GOT_PAYLOAD:
00127     if (c != gps_skytraq.checksum) {
00128       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_CHECKSUM;
00129       goto error;
00130     }
00131     gps_skytraq.status = GOT_CHECKSUM;
00132     break;
00133   case GOT_CHECKSUM:
00134     if (c != SKYTRAQ_SYNC3) {
00135       gps_skytraq.error_last = GPS_SKYTRAQ_ERR_OUT_OF_SYNC;
00136       goto error;
00137     }
00138     gps_skytraq.status = GOT_SYNC3;
00139     break;
00140   case GOT_SYNC3:
00141     gps_skytraq.msg_available = TRUE;
00142     goto restart;
00143   default:
00144     gps_skytraq.error_last = GPS_SKYTRAQ_ERR_UNEXPECTED;
00145     goto error;
00146   }
00147   return;
00148  error:
00149   gps_skytraq.error_cnt++;
00150  restart:
00151   gps_skytraq.status = UNINIT;
00152   return;
