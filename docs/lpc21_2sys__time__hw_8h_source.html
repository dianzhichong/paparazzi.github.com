00092 static inline void sys_time_init( void ) {
00093   /* setup Timer 0 to count forever  */
00094   /* reset & disable timer 0         */
00095   T0TCR = TCR_RESET;
00096   /* set the prescale divider        */
00097   T0PR = T0_PCLK_DIV - 1;
00098   /* disable match registers         */
00099   T0MCR = 0;
00100   /* disable compare registers       */
00101   T0CCR = 0;
00102   /* disable external match register */
00103   T0EMR = 0;
00104   /* enable timer 0                  */
00105   T0TCR = TCR_ENABLE;
00106 
00107   cpu_time_sec = 0;
00108   cpu_time_ticks = 0;
00109 
00110   /* select TIMER0 as IRQ    */
00111   VICIntSelect &= ~VIC_BIT(VIC_TIMER0);
00112   /* enable TIMER0 interrupt */
00113   VICIntEnable = VIC_BIT(VIC_TIMER0);

00117   _VIC_ADDR(TIMER0_VIC_SLOT) = (uint32_t)TIMER0_ISR;

00138 static inline bool_t sys_time_periodic( void ) {
00139   uint32_t now = T0TC;
00140   uint32_t dif = now - last_periodic_event;
00141   if ( dif >= PERIODIC_TASK_PERIOD) {
00142     last_periodic_event += PERIODIC_TASK_PERIOD;
00143     cpu_time_ticks += PERIODIC_TASK_PERIOD;
00144     if (cpu_time_ticks > TIME_TICKS_PER_SEC) {
00145       cpu_time_ticks -= TIME_TICKS_PER_SEC;
00146       cpu_time_sec++;
00147 #ifdef TIME_LED
00148       LED_TOGGLE(TIME_LED)
00149 #endif
00150 #ifdef SYS_TIME_LED
00151       LED_TOGGLE(SYS_TIME_LED)
00152 #endif
00153     }
00154     return TRUE;
00155   }
00156   return FALSE;
00157 }
